```js
1.var 关键字
for(var i = 0; i < 5; i++){
    setTimeout(() =>{
    console.log(i)
    },1000)
}
// > 5 5 5 5 5

2.let 形成局部作用域
for(let i=0;i<5;i++){
    setTimeout(()=>console.log(i),i\*1000)
}
// >0 1 2 3 4

3.setTimeout 第三个参数为传入参数
    for(var i=0;i<5;i++){
    setTimeout((j)=>console.log(j),i\*1000,i)
}
// >0 1 2 3 4
```
